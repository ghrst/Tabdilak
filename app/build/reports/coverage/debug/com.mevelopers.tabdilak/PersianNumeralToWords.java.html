	public static String numberToString(long number) {
		if(number == 0) {
			return "طµظپط±";
		}
		/*
		 * Notice that in here we ignore sign of input number.
		 */
		number = Math.abs(number);
		String sNumber = Long.toString(number);
		//Check to see if number is in our legam range
		if(sNumber.length() > 12) {
			throw new NumberFormatException("Number is too large for parser!");
		}
		//We must pad the leftmost side with zeros
		String mask = "000000000000";
		DecimalFormat df = new DecimalFormat(mask);
		sNumber = df.format(number);
		//XXXnnnnnnnnn
		int billions = Integer.parseInt(sNumber.substring(0, 3));
		//nnnXXXnnnnnn
		int millions = Integer.parseInt(sNumber.substring(3, 6));
		//nnnnnnXXXnnn
		int hundredThousands = Integer.parseInt(sNumber.substring(6,9));
		//nnnnnnnnnXXX
		int thousands = Integer.parseInt(sNumber.substring(9,12));
		
		String billionsString = convertHelper(billions);
		
		String millionsString = convertHelper(millions);
		
		String hundredThousandsString = convertHelper(hundredThousands);
		
		String thousandsString = convertHelper(thousands);
		
		String result = "";
		boolean addAnd = false;
		if(billionsString != "") {
			result += billionsString + " ظ…غŒظ„غŒط§ط±ط¯";
			addAnd = true;
		}
		
		if(millionsString != "") {
			if(addAnd) {
				result += " ظˆ " + millionsString + " ظ…غŒظ„غŒظˆظ†";
			} else {
				result += millionsString + " ظ…غŒظ„غŒظˆظ†";
			}
			addAnd = true;
		}
		
		if(hundredThousandsString != "") {
			if(addAnd) {
				result += " ظˆ " + hundredThousandsString + " ظ‡ط²ط§ط±";
			} else {
				result += hundredThousandsString + " ظ‡ط²ط§ط±";
			}
			addAnd = true;
		}
		
		if(thousandsString != "") {
			if(addAnd) {
				result += " ظˆ " + thousandsString;
			} else {
				result += thousandsString;
			}
		}
		
		return result;
	}
	
	public static String convertHelper (int number) {
		String result = "";
		boolean addAnd = false;
		int hundreds = number / 100;
		if(hundreds > 0) {
		
			result += mHundreds[hundreds];
			addAnd = true;
		}
		number = number % 100;
		if(number < 20 && number > 0){
			if(addAnd) {
				result += " ظˆ " + mOnes[number];
			} else {
				result += mOnes[number];
			}
		}
		
		if(number >= 20) {
			int tens = number / 10;
			if(addAnd) {
				result += " ظˆ " + mTens[tens];
				number = number % 10;
				if(number > 0)
					result += " ظˆ " + mOnes[number];
			} else {
				result += mTens[tens];
				number = number % 10;
				if(number > 0) {
					result += " ظˆ " + mOnes[number];
				}
			}	
		}
		return result;
	}
